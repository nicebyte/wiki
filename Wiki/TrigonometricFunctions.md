$$
\sin{θ} = \frac{b}{c}
$$
$$
\cos{θ} = \frac{a}{c}
$$
$$
\tan{θ} = \frac{b}{a} = \frac{\sin{θ}}{\cos{θ}}
$$
$$
\cot{θ} = \frac{a}{b} = \frac{\cos{θ}}{\sin{θ}}
$$
$$
\sin^2{θ} + \cos^2{θ} = 1
$$
